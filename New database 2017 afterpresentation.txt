CREATE DATABASE studentrecordkeeper;

CREATE TABLE courses
(
  coursesubject text,
  coursenumber text,
  coursesection text,
  coursename text,
  credits double precision,
  yearlevel integer,
  yearsection text,  
  yearused integer,  
  PRIMARY KEY (coursesubject,coursenumber)
);

CREATE TABLE student
(
  name text,
  studentnumber text PRIMARY KEY,
  previouschool text,
  previousprogram text,
  yearlevel integer,
  yearsection text,  
  year integer
);

CREATE TABLE grades
{
	studentnumber text REFRENCES student ON DELETE CASCADE,
	coursesubject text REFERENCES courses ON DELETE RESTRICT,
	coursenumber text REFERENCES courses ON DELETE RESTRICT,
	grade integer,
	yeartaken integer,
	PRIMARY KEY(studentnumber,coursenumber,coursesubject,yeartaken)
};
